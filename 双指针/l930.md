package l930;

public class Solution930 {
    public int numSubarraysWithSum(int[] A, int S) {
        int length = A.length;
        if (length == 0) {
            return 0;
        }

        int[] sumArray = new int[length];
        int sum = 0;
        for (int i = 0; i < length; i++) {
            sum += A[i];
            sumArray[i] = sum;
        }

        int count = 0;
        for (int i = 0; i < length; i++) {
            for (int j = i; j < length; j++) {
                // [i,j]
                int base = i < 1 ? 0 : sumArray[i - 1];
                int curSum = sumArray[j] - base;
                if (curSum == S) {
                    count++;
                } 
                //这里稍微优化了下才过了
                else if (curSum > S) {
                    break;
                }

            }
        }

        return count;
    }
}
